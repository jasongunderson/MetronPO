CREATE TABLE users(
uid CHAR(21) PRIMARY KEY,
first CHAR(20),
last CHAR(20),
pass CHAR(62),
facility CHAR(20),
admin INT
);


CREATE TABLE requests(
rid INT PRIMARY KEY,
uid CHAR(21),
facility CHAR(20),
department CHAR(20),
vendor CHAR(32),
description CHAR(32),
cost DECIMAL(10,2),
quantity INT,
reason CHAR(255),
date_submit DATE,
date_needed DATE,
admin DATE,
coo DATE,
cfo DATE
);
